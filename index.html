<html>

<head>

    <style>

        body {

            font-family: Arial, sans-serif;

            font-size: 24px;

            text-align: center;

            padding: 20px;

        }

        #text {

            border-right: 3px solid black;

            padding-right: 5px;

        }

    </style>

</head>

<body>

    <span id="text"></span>

    <script>

        const texts = [

            "I am a Business Analyst enthusiast.",

            "I am a Data Analyst enthusiast.",

            "I am a Digital Transformation enthusiast."

        ];

        let index = 0;

        let letterIndex = 0;

        let currentText = "";

        let isDeleting = false;


        function type() {

            if (!isDeleting && letterIndex < texts[index].length) {

                currentText = texts[index].slice(0, ++letterIndex);

            } else if (isDeleting && letterIndex > 0) {

                currentText = texts[index].slice(0, --letterIndex);

            }


            document.getElementById("text").textContent = currentText;


            let speed = isDeleting ? 50 : 100;

            if (!isDeleting && letterIndex === texts[index].length) {

                speed = 1500;

                isDeleting = true;

            } else if (isDeleting && letterIndex === 0) {

                isDeleting = false;

                index = (index + 1) % texts.length;

                speed = 500;

            }


            setTimeout(type, speed);

        }


        type();

    </script>

</body>

</html>
